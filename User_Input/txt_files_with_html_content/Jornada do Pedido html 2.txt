    <script>
        // Quantity input handling
        const decreaseBtn = document.getElementById('decrease-btn');
        const increaseBtn = document.getElementById('increase-btn');
        const quantityInput = document.getElementById('quantity-input');
        const confirmBtn = document.getElementById('confirm-btn');
        const progressBar = document.querySelector('.bg-primary-500');
        
        // Update button state based on quantity
        function updateButtonState() {
            const quantity = parseInt(quantityInput.value);
            confirmBtn.disabled = quantity === 0;
            
            // Update progress bar
            const expectedQuantity = 5; // This would come from your data
            const percentage = Math.min(100, (quantity / expectedQuantity) * 100);
            progressBar.style.width = `${percentage}%`;
        }
        
        decreaseBtn.addEventListener('click', () => {
            const currentValue = parseInt(quantityInput.value);
            if (currentValue > 0) {
                quantityInput.value = currentValue - 1;
                updateButtonState();
            }
        });
        
        increaseBtn.addEventListener('click', () => {
            const currentValue = parseInt(quantityInput.value);
            quantityInput.value = currentValue + 1;
            updateButtonState();
        });
        
        quantityInput.addEventListener('change', () => {
            // Ensure value is not negative
            if (parseInt(quantityInput.value) < 0) {
                quantityInput.value = 0;
            }
            updateButtonState();
        });
        
        // Confirm button action
        confirmBtn.addEventListener('click', () => {
            // Here you would typically send the data to your backend
            // and then either load the next product or move to the next stage
            
            // For demonstration, let's show a success message
            alert('Produto confirmado! Avançando para o próximo item...');
            
            // Reset for next product (in a real app, you'd load new product data)
            quantityInput.value = 0;
            updateButtonState();
        });
        
        // Initialize button state
        updateButtonState();
    </script>
