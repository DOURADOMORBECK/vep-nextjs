    <script>
        // Quantity counter functionality
        const decreaseBtn = document.getElementById('decrease-btn');
        const increaseBtn = document.getElementById('increase-btn');
        const quantityDisplay = document.getElementById('quantity-display');
        let quantity = 0;
        
        decreaseBtn.addEventListener('click', () => {
            if (quantity > 0) {
                quantity--;
                quantityDisplay.textContent = quantity;
                updateButtonStates();
            }
        });
        
        increaseBtn.addEventListener('click', () => {
            quantity++;
            quantityDisplay.textContent = quantity;
            updateButtonStates();
        });
        
        function updateButtonStates() {
            if (quantity === 0) {
                decreaseBtn.classList.add('opacity-50');
            } else {
                decreaseBtn.classList.remove('opacity-50');
            }
            
            // Highlight confirm button when quantity matches expected
            const confirmBtn = document.getElementById('confirm-btn');
            if (quantity === 12) {
                confirmBtn.classList.remove('bg-primary-600');
                confirmBtn.classList.add('bg-primary-500', 'animate-pulse');
            } else {
                confirmBtn.classList.add('bg-primary-600');
                confirmBtn.classList.remove('bg-primary-500', 'animate-pulse');
            }
        }
        
        // Initialize button states
        updateButtonStates();
        
        // Simulate navigation
        document.getElementById('confirm-btn').addEventListener('click', function() {
            if (quantity > 0) {
                alert('Produto separado com sucesso! Avançando para o próximo item...');
            } else {
                alert('Por favor, insira a quantidade separada.');
            }
        });
        
        document.getElementById('skip-btn').addEventListener('click', function() {
            alert('Item pulado. Será necessário revisão posterior.');
        });
        
        document.getElementById('back-button').addEventListener('click', function() {
            alert('Deseja realmente sair da jornada? O progresso atual será perdido.');
        });
    </script>
