    <script>
        // Toggle password visibility
        document.getElementById('toggle-password').addEventListener('click', function() {
            const passwordInput = document.getElementById('password');
            const icon = this.querySelector('i');
            
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';
                icon.classList.remove('fa-eye');
                icon.classList.add('fa-eye-slash');
            } else {
                passwordInput.type = 'password';
                icon.classList.remove('fa-eye-slash');
                icon.classList.add('fa-eye');
            }
        });
        
        // User type selection
        const userTypeElements = document.querySelectorAll('.user-type');
        userTypeElements.forEach(element => {
            element.addEventListener('click', function() {
                // Remove active class from all
                userTypeElements.forEach(el => {
                    el.classList.remove('border-primary-500', 'bg-dark-600');
                    el.querySelector('i').classList.remove('text-primary-500');
                    el.querySelector('i').classList.add('text-gray-300');
                });
                
                // Add active class to selected
                this.classList.add('border-primary-500', 'bg-dark-600');
                this.querySelector('i').classList.remove('text-gray-300');
                this.querySelector('i').classList.add('text-primary-500');
            });
        });
        
        // Form submission
        document.getElementById('login-form').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Get selected user type
            let selectedUserType = null;
            userTypeElements.forEach(el => {
                if (el.classList.contains('border-primary-500')) {
                    selectedUserType = el.getAttribute('data-type');
                }
            });
            
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            
            if (!username || !password || !selectedUserType) {
                // Show error (in a real app)
                alert('Por favor, preencha todos os campos e selecione um tipo de usuário');
                return;
            }
            
            // Simulate login - in a real app, this would be an API call
            console.log('Login attempt:', {
                username,
                password: '********',
                userType: selectedUserType
            });
            
            // Redirect to dashboard (simulated)
            alert(`Login bem-sucedido como ${selectedUserType}! Redirecionando para o Dashboard...`);
        });
    </script>
