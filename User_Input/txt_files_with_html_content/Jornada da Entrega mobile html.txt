    <script>
        // Initialize Leaflet map
        document.addEventListener('DOMContentLoaded', function() {
            const map = L.map('map-container').setView([-23.550520, -46.633308], 12);
            
            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                attribution: '&copy; OpenStreetMap contributors'
            }).addTo(map);
            
            // Add markers for delivery points
            const warehouse = [-23.545000, -46.630000];
            const deliveryPoints = [
                [-23.555000, -46.640000],
                [-23.560000, -46.635000],
                [-23.550000, -46.645000],
                [-23.540000, -46.638000]
            ];
            
            // Warehouse marker
            L.marker(warehouse).addTo(map)
                .bindPopup('Depósito')
                .openPopup();
            
            // Delivery point markers
            deliveryPoints.forEach((point, index) => {
                L.marker(point).addTo(map)
                    .bindPopup(`Cliente ${index + 1}`);
            });
            
            // Draw route line
            const routePoints = [warehouse, ...deliveryPoints, warehouse];
            L.polyline(routePoints, {color: '#16a34a'}).addTo(map);
            
            // Fit bounds to show all markers
            const bounds = L.latLngBounds([warehouse, ...deliveryPoints]);
            map.fitBounds(bounds);
        });
        
        // Handle vehicle selection
        const vehicleCards = document.querySelectorAll('#vehicle-selection > div');
        vehicleCards.forEach(card => {
            card.addEventListener('click', function() {
                vehicleCards.forEach(c => {
                    c.classList.remove('border-primary-500');
                    c.querySelector('i').classList.remove('text-primary-500');
                    c.querySelector('i').classList.add('text-gray-400');
                });
                
                this.classList.add('border-primary-500');
                this.querySelector('i').classList.remove('text-gray-400');
                this.querySelector('i').classList.add('text-primary-500');
            });
        });
        
        // Handle operator selection
        const operatorCards = document.querySelectorAll('#operator-selection > div');
        operatorCards.forEach(card => {
            card.addEventListener('click', function() {
                operatorCards.forEach(c => {
                    c.classList.remove('border-primary-500');
                    c.querySelector('i').classList.remove('text-primary-500');
                    c.querySelector('i').classList.add('text-gray-400');
                    
                    // Remove check icon if exists
                    const checkIcon = c.querySelector('.fa-check');
                    if (checkIcon) {
                        checkIcon.remove();
                    }
                });
                
                this.classList.add('border-primary-500');
                this.querySelector('i').classList.remove('text-gray-400');
                this.querySelector('i').classList.add('text-primary-500');
                
                // Add check icon if not exists
                if (!this.querySelector('.fa-check')) {
                    const checkIcon = document.createElement('i');
                    checkIcon.className = 'fa-solid fa-check text-primary-500';
                    this.appendChild(checkIcon);
                }
            });
        });
        
        // Navigation buttons
        document.getElementById('back-step-button').addEventListener('click', function() {
            alert('Voltando para a etapa de Rotas');
        });
        
        document.getElementById('next-step-button').addEventListener('click', function() {
            alert('Avançando para a etapa de Mapa');
        });
    </script>
